CREATE TABLE Reserva (
	ID INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    dataReserva DATETIME NOT NULL,
    horarioID INT NOT NULL,
    FOREIGN KEY (horarioID) REFERENCES horarios (ID),
    centroEsportivo INT NOT NULL,
    FOREIGN KEY (centroEsportivo) REFERENCES centrosesportivos (ID),
    alunoResponsavel INT NOT NULL,
    FOREIGN KEY (alunoResponsavel) REFERENCES Alunos (RA),
    created DATETIME NOT NULL,
    deleted DATETIME
);

CREATE TABLE ParticipantesReserva (
	aluno INT NOT NULL,
    reserva INT NOT NULL,
    PRIMARY KEY (aluno, reserva),
	FOREIGN KEY (aluno) REFERENCES Alunos (RA),
	FOREIGN KEY (reserva) REFERENCES Reserva (ID),
    created DATETIME NOT NULL,
    deleted DATETIME
);
